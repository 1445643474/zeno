file(GLOB dirs LIST_DIRECTORIES TRUE CONFIGURE_DEPENDS *)

foreach (dir ${dirs})
    if (IS_DIRECTORY ${dir})
        get_filename_component(name ${dir} NAME)
        option(ZENO_WITH_${name} "Build Zeno module ${name}" ON)
        message("-- Zeno module ${name} is [${ZENO_WITH_${name}}]")
        if (ZENO_WITH_${name})
            add_subdirectory(${name})
        endif()
    endif()
endforeach()
